	public void ChooseNumberOfRoom(int numberRoom) {
		String RoomPath = "/hierarchy/android.widget.FrameLayout/android.widget.LinearLayout/android.widget.FrameLayout/android.widget.FrameLayout/android.widget.FrameLayout/android.view.View/android.view.View/android.view.View[1]/android.view.View/android.widget.Button[2]";
		WebElement Room = driver.findElement(MobileBy.xpath(RoomPath));

		if (numberRoom <= 7) {
			for (int i = 1; i < numberRoom; i++) {
				Room.click();
			}
			System.out.println(Room.toString());
		} else {
			System.out.println(" Just only choose maximum 7 Room in Hotel");
		}
	}
